🔍 Trong KBLab LMS: Core Service gọi Judge Service qua OpenFeign để chấm bài SQL. Nếu Judge Service chết hoặc quá chậm, Circuit Breaker ngăn Core Service treo theo — trả lỗi "Judge unavailable" ngay lập tức thay vì chờ timeout 30 giây.
SqlExecutorService.java 
// Resilience4j Circuit Breaker — KBLab Judge Service
@CircuitBreaker(name = "judgeService", fallbackMethod = "judgeFallback")
@Retry(name = "judgeService")
public JudgeResult executeSQL(Submission submission) {
    return judgeFeignClient.judge(submission);
}

private JudgeResult judgeFallback(Submission s, Exception e) {
    log.warn("Judge unavailable, queuing for retry", e);
    kafkaProducer.sendToRetryQueue(s);
    return JudgeResult.pending("Judge tạm không khả dụng");
}
